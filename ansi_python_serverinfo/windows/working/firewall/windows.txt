- name: test firewall on Windows
  hosts: "{{ servers }}"
  vars:
    destinations: ""  # Provide the string of IP addresses
    ports: ""         # Provide the string of ports
    protocol: ""      # The protocol variable from Ansible Tower

  tasks:
    - name: Split IP and Port Strings
      set_fact:
        destinations_list: "{{ destinations.split(',') | join(' ').split(' ') }}"
        ports_list: "{{ ports.split(',') | join(' ').split(' ') }}"

    - name: Combine IP and Port Pairs
      set_fact:
        combined_destinations: "{{ combined_destinations | default([]) + [{'ip': item[0], 'port': item[1]}] }}"
      loop: "{{ destinations_list | zip(ports_list) | list }}"
      register: combined_results

    - name: Execute PowerShell command
      script: |
        $ErrorActionPreference = "Stop"
        $results = @()
        foreach ($item in $using:combined_results.results) {
          if ("{{ protocol }}" -eq "tcp") {
            $result = Test-NetConnection -ComputerName $item.item.ip -Port $item.item.port
            $status = if ($result.TcpTestSucceeded) { "success" } else { "fail" }
          } else {
            $result = Test-NetUDPPort -ComputerName $item.item.ip -Port $item.item.port
            $status = if ($result.UdpTestSucceeded) { "success" } else { "fail" }
          }
          $output = "{0} {1} {2}" -f $item.item.ip, $item.item.port, $status
          $results += $output
        }
        $results
      loop: "{{ combined_results.results }}"
      register: con

    - debug:
        var: item.stdout_lines
      loop: "{{ con.results }}"
