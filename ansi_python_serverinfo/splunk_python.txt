urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# Splunk API endpoint
splunk_url = "https://your-splunk-server:8089/services/search/jobs"

# Splunk search query
search_query = ' (index=test_index) (Source="/app/access.*")  (NOT "_raw" =*/monitor/health.html*") | search finalstatus=500 | stats dedup_splitvals=t count  As Count by remotehost | fields - _span |  fillnull Count | fields remotehost, Count | sort - Count'

# Splunk time range for the last 5 minutes
time_range = "-5m"

# Splunk authentication details
username = "your-username"
password = "your-password"

# Define headers for Splunk API request
headers = {
    "Content-Type": "application/x-www-form-urlencoded",
    "Accept": "application/json"
}

# Prepare Splunk API request payload
payload = {
    "search": search_query,
    "earliest_time": time_range,
    "output_mode": "csv"
}

# Send request to Splunk API with SSL verification disabled
response = requests.post(splunk_url, data=payload, headers=headers, auth=(username, password), verify=False)

# Check response status code

    # Retrieve the search ID from the response
    search_id = response.json()["sid"]
    print("Search ID:", search_id)

    # Poll the search job until it is complete

        "output_mode": "json"
